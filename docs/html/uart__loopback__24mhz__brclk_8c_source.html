187void EUSCIA2_IRQHandler(void){
188 uint32_t status = MAP_UART_getEnabledInterruptStatus(EUSCI_A2_BASE);
189
190 if(status & EUSCI_A_UART_RECEIVE_INTERRUPT_FLAG){
191 RXData = MAP_UART_receiveData(EUSCI_A2_BASE);
192 if(dataCount < RX_BUFFER_SIZE){
193 data[dataCount++] = RXData;
194 }else if(dataCount > RX_BUFFER_SIZE - 1 || (RXData == '\r' && data[dataCount-1] == '\n')){
195 stringEnd = true;
196 MAP_Interrupt_disableSleepOnIsrExit();
197 }
198 }
199}
